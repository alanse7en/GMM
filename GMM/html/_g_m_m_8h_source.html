21 struct fitOption {
22  /*
23  covType : 'diagonal' covariance matrices are restricted to be diagonal;
24  'spherical' covariance matrices are restricted to be spherical;
25  'full' covariance matrices are restricted to be full( default).
26  sharedCov : True if the component covariance matrices are restricted to be the same, default is false.
27  start : 'random' randomly select from train data to initialize model parameters;
28  'kmeans' use kmeans to get initial clustering index of the train data( default).
29  regularize : Regularization parameter to avoid covariance being singular, defalut is 0.
30  display : 'iter' show information during training (default); 'final' show final information.
31  maxIter : Maximum iterations allowed, default is 100;
32  tolFun : The termination tolerance for the log-likelihood function, default is 1e-6.
33  converged : 'true' GMM training process converged; otherwise failed.
34  iters : The num of iterations to finished training.
35  */
36  string covType = "full";
37  bool sharedCov = false;
38  string start = "kmeans";
39  double regularize = 0.0;
40  string display = "iter";
41  unsigned int maxIter = 100;
42  double tolFun = 1e-6;
43  bool converged = false;
44  unsigned int iters = 0;
45 };
